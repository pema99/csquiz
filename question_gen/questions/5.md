using System;

public class Program
{
    public static void Foo<T>()
        where T : new()
    {
        T t = new T();
        if (typeof(T) == t.GetType())
            Console.Write("0");
        else
            Console.Write("1");
    }
    
    public static void Main()
    {
        try
        {
            Foo<int>();
            Foo<int?>();
        }
        catch
        {
            Console.Write("2");
        }
    }
}
